{{ func render_enum(v)
    if object.typeof(v) == 'array'
        $result = ''
        $index = 0
        if v.size > 1
            $result += '('
        end
        
        for x in v
            $index = $index + 1
            $result += render_enum(x)

            if $index < v.size
                $result += ' '
            end
        end
        
        if v.size > 1
            $result += ')'
        end

        ret $result
    else
        ret v
    end
end }}

template <>
{{ item.name }} enum_parse<{{ item.name }}>(const std::string k)
{
    static const std::unordered_map<std::string, {{ item.name }}> enums
    {
{{~ for prop in item.props ~}}
        { "{{ prop.name }}", {{ item.name }}::{{ prop.name }} }{{ if !for.last }}, {{ end }}
{{~ end ~}}
    };

    auto i = enums.find(k);
    if (i == enums.end())
        throw std::runtime_error("no enum value");

    return i->second;
}
