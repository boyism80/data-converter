template <typename K, typename V>
fb::model::kv_container<K, V>::kv_container(const std::string& fname) : _fname(fname)

fb::model::kv_container<K, V>::kv_container(const Json::Value& json)
{ }

template <typename K, typename V>
void fb::model::kv_container<K, V>::load(const std::string& fname)
{
    Json::Value             json;
    std::ifstream           ifstream;
    ifstream.open(fname);
    if (ifstream.is_open() == false)
        throw std::runtime_error("cannot find json file");

    Json::Reader            reader;
    if (reader.parse(ifstream, json) == false)
    {
        ifstream.close();
        throw std::runtime_error("cannot parse json file");
    }

    ifstream.close();
    this->load(json);
}

template <typename K, typename V>
void fb::model::kv_container<K, V>::load(const Json::Value& json)
{
    for (auto i = json.begin(); i != json.end(); i++)
    {
        std::map<K, V>::insert({ fb::model::build<K>(i.key()), fb::model::build<V>(*i) });
    }
}

template <typename K, typename V>
void fb::model::kv_container<K, V>::load()
{
    this->load(this->_fname);
}

template <typename T>
void fb::model::array_container<T>::load(const std::string& fname)
{
    Json::Value             json;
    std::ifstream           ifstream;
    ifstream.open(fname);
    if (ifstream.is_open() == false)
        throw std::runtime_error("cannot find json file");

    Json::Reader            reader;
    if (reader.parse(ifstream, json) == false)
    {
        ifstream.close();
        throw std::runtime_error("cannot parse json file");
    }

    ifstream.close();
    this->load(json);
}

template <typename T>
void fb::model::array_container<T>::load(const Json::Value& json)
{
    for (auto i = json.begin(); i != json.end(); i++)
    {
        std::vector<T>::push_back(fb::model::build<T>(*i));
    }
}

template <typename T>
void fb::model::array_container<T>::load()
{
    this->load(this->_fname);
}
