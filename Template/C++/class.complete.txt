namespace fb { namespace model {

template<class T>
struct is_optional : std::false_type {};

template<class T>
struct is_optional<std::optional<T>> : std::true_type {};

template<class T>
struct is_vector : std::false_type {};

template<class T>
struct is_vector<std::vector<T>> : std::true_type {};

template <typename T, typename Enabler = void>
struct is_map : std::false_type {};

template <typename T>
struct is_map<T, std::void_t<typename T::mapped_type>> : std::true_type{};

template <typename T>
struct is_default
{
    static constexpr bool value =
        !std::is_enum<T>::value &&
        !fb::model::is_vector<T>::value &&
        !fb::model::is_map<T>::value &&
        !fb::model::is_optional<T>::value;
};

class dsl
{};

class date_range
{
public:
    const boost::posix_time::ptime begin, end;

public:
    date_range(const boost::posix_time::ptime& begin, const boost::posix_time::ptime& end) : 
        begin(begin),
        end(end)
    {}
    ~date_range() = default;
};

class base
{
protected:
    base() = default;
public:
    ~base() = default;

public:
    template <typename T>
    typename std::enable_if<fb::model::is_default<T>::value, T>::type build(const Json::Value& json)
    {
        throw std::runtime_error("does not support type");
    }

    template <typename T>
    typename std::enable_if<std::is_enum<T>::value, T>::type build(const Json::Value& json)
    {
        return T(json.asInt());
    }

    template <typename T>
    typename std::enable_if<fb::model::is_vector<T>::value, T>::type build(const Json::Value& json)
    {
        auto result = std::vector<typename T::value_type>();
        for (auto i = json.begin(); i != json.end(); i++)
        {
            result.push_back(build<typename T::value_type>(*i));
        }
        return result;
    }

    template <typename T>
    typename std::enable_if<fb::model::is_map<T>::value, T>::type build(const Json::Value& json)
    {
        return std::map<typename T::key_type, typename T::mapped_type>();
    }

    template <typename T>
    typename std::enable_if<fb::model::is_optional<T>::value, T>::type build(const Json::Value& json)
    {
        if (json.isNull())
            return std::nullopt;
        else
            return std::optional<typename T::value_type>(build<T::value_type>(json));
    }

    template <>
    fb::model::dsl build<fb::model::dsl>(const Json::Value& json)
    {
        return fb::model::dsl();
    }

    template <>
    int build<int>(const Json::Value& json)
    {
        return json.asInt();
    }

    template <>
    std::string build<std::string>(const Json::Value& json)
    {
        if(json.isNull())
            return "";
        else
            return json.asString();
    }

    template <>
    float build<float>(const Json::Value& json)
    {
        return json.asFloat();
    }

    template <>
    double build<double>(const Json::Value& json)
    {
        return json.asDouble();
    }

    template <>
    long build<long>(const Json::Value& json)
    {
        return json.asLargestInt();
    }

    template <>
    bool build<bool>(const Json::Value& json)
    {
        return json.asBool();
    }

    template <>
    boost::posix_time::ptime build<boost::posix_time::ptime>(const Json::Value& json)
    {
        return boost::posix_time::time_from_string(json.asCString());
    }

    template <>
    std::chrono::milliseconds build<std::chrono::milliseconds>(const Json::Value& json)
    {
        static const auto regex = boost::xpressive::sregex::compile("(?P<hour>\\d{1,2}):(?P<min>\\d{1,2}):(?P<sec>\\d{1,2})(?:.(?P<ms>\\d+))?");
        auto what = boost::xpressive::smatch();
        auto data = json.asString();
        if (boost::xpressive::regex_match(data, what, regex) == false)
            throw std::runtime_error("cannot parse timespan");

        auto hours = std::stoi(what["hour"].str());
        auto mins = std::stoi(what["min"].str());
        auto secs = std::stoi(what["sec"].str());
        auto ms = what["ms"].matched ? std::stoi(what["ms"].str()) : 0;

        return std::chrono::milliseconds(ms + (secs * 1000) + (mins * 60 * 1000) + (hours * 60 * 60 * 1000));
    }

    template <>
    fb::model::date_range build<fb::model::date_range>(const Json::Value& json)
    {
        return fb::model::date_range(build<boost::posix_time::ptime>(json["Begin"]), build<boost::posix_time::ptime>(json["End"]));
    }
};
